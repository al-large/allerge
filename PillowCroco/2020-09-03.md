- 그래서 dp로 선회하려 했는데, 생각해보니 `-20~+20`까지 정도라면 그냥 전체 탐색(brute force)해도 큰문제 없을 것 같음
- 모듈화를 통해 디버그를 잘 했으나, 삽질해서 시간이 오래 걸린 케이스.
- `for`문 돌리는거 범위 체크 실수로 시간 날려먹었다 이런~

```c++
vvi rotate(vvi key) {
    vvi rot = key;
    int size = key.size();
    for (int i = 0; i < size; i++) {
        for (int j = 0; j < size; j++) {
            rot[j][size - i - 1] = key[i][j];
        }
    }
    return rot;
}

//x, y 는 절댓값이 key size를 초과하지 않는다.
vvi push(vvi key, int x, int y) {
    vvi pus = key;
    int size = key.size();
    if (x > 0) {
        for (int i = size - 1; i >= x; i--)
            for (int j = 0; j < size; j++)
                pus[i][j] = key[i - x][j];

        for (int i = 0; i < x; i++)
            for (int j = 0; j < size; j++)
                pus[i][j] = 0;
    }
    else if (x < 0) {
        for (int i = 0; i < size + x; i++)
            for (int j = 0; j < size; j++)
                pus[i][j] = key[i - x][j];

        for (int i = size + x; i < size; i++)
            for (int j = 0; j < size; j++)
                pus[i][j] = 0;
    }

    vvi ret = pus;

    if (y > 0) {
        for (int i = size - 1; i >= y; i--)
            for (int j = 0; j < size; j++)
                ret[j][i] = pus[j][i - y];

        for (int i = 0; i < y; i++)
            for (int j = 0; j < size; j++)
                ret[j][i] = 0;
    }
    else if (y < 0) {
        for (int i = 0; i < size + y; i++)
            for (int j = 0; j < size; j++)
                ret[j][i] = pus[j][i - y];

        for (int i = size + y; i < size; i++)
            for (int j = 0; j < size; j++)
                ret[j][i] = 0;
    }

    return ret;
}

int getSum(vvi key) {
    int size = key.size();
    int sum = 0;
    for (int i = 0; i < size; i++)
        for (int j = 0; j < size; j++)
            sum += key[i][j];
    return sum;
}

            ret &= bool(key[i][j] ^ lock[i][j]);
    return ret;
}
```

```c++
class Solution {
public:
    vector<string> letterCombinations(string digits) {
        if(digits.empty())
            return vector<string>();

        //mapping num to alpha
        //string으로 짜면 되네.. 개바보다.
        // map<char,vector<char>> m;
        // for(int i=2;i<=6;i++)
        //     for(int j=0;j<3;j++)
        //         m[char(i+'0')].push_back(char('a'+3*(i-2)+j));
        // vector<char> v7{'p','q','r','s'}, v8{'t','u','v'}, v9{'w','x','y','z'};
        // m['7'] = v7;
        // m['8']=v8;
        // m['9']=v9;

        string m[8] {"abc","edf","ghi","jkl","mno","pqrs","tuv","wxyz"};

        vector<string> cur(1,"");
        vector<string> next;
        for(auto c:digits){
            for(int i=0;i<m[c-'2'].size();i++){
                for(int j=0;j<cur.size();j++){
                    string s = cur[j];
                    s += m[c-'2'][i];
                    next.push_back(s);
                }
            }
            cur=next;
            next.clear();
        }

        return cur;
    }
};
```

### 49. Group Anagrams
https://leetcode.com/problems/group-anagrams/

개꿀


```c++
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        unordered_map<string, vector<string>> dict;


        for(auto str: strs){
            vector<int> char_cnt (26,0);
            for(auto c: str){
                char_cnt[c-'a']++;
            }

            string s="";
            for(int i=0;i<26;i++){
                for(int j=0;j<char_cnt[i];j++){
                    s += char(i+'a');
                }
            }
            dict[s].push_back(str);
        }

        vector<vector<string>> ans(dict.size(), vector<string>());

        int i=0;
        for(auto d: dict){
            for(auto s: d.second)
                ans[i].push_back(s);
            i++;
        }

        return ans;
    }
};
```
