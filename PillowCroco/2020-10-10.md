```c++
#if 1
#define DEB 0
#include<iostream>
#include<queue>
using namespace std;

int n, num;

// ma는 median 이하의 maximum, mi는 median 초과의 minimum
priority_queue<int, vector<int>, less<int>> ma;
priority_queue<int, vector<int>, greater<int>> mi;
int main() {
	cin.sync_with_stdio(false);
	cin.tie(nullptr);
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> num;

		//first entry
		if (ma.empty()) {
			ma.push(num);
			cout << ma.top() << '\n';
			continue;
		}

		if (ma.size() > mi.size()) {
			if (num < ma.top()) {
				mi.push(ma.top());
				ma.pop();
				ma.push(num);
			}
			else {
				mi.push(num);
			}
		}

		// ma.size() == mi.size()
		else {
			if (num > mi.top()) {
				ma.push(mi.top());
				mi.pop();
				mi.push(num);
			}
			else {
				ma.push(num);
			}
		}
#if DEB
		cout << "ma.top(): " << ma.top() << ", ma.size(): " << ma.size() << endl;
		cout << "mi.top(): " << mi.top() << ", mi.size(): " << mi.size() << endl;
#endif
		cout << ma.top() << '\n';

	}
	return 0;

}
#endif
```
