### 부분수열의 합 2
https://www.acmicpc.net/problem/1208

- map 사용하니까 TLE 뜸. map에 사용된 애를 vector로 복사하는 과정에서 시간초과가 난듯.
- 메모리 많이 써도 이렇게 하는게 더 나은가보다!
- __MITM 알고리즘__, 처음해봤다.
- 조합짜기는 dfs를 통해서 할 수 있다. 굳 테크닉

```c++
#if 1
#define DEB 0
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

// 2^40 돌리면 정신나가므로 중간에 쪼개야함 -> 이게 가운데에서 만나기, MITM알고리즘임
// 2^20 * 2로 확인한다.
// 2^20은 10^6정도로, 100만이면 1초내로 충분히가능한 수준.


// 절반을 어떻게 쪼개느냐? 왼쪽 반과 오른쪽 반을 다른 성질로 하면 됨.
// 왼쪽 절반으로 만드는 값이 a이고, 오른쪽 절반으로 만드는값이 b이면
// a+b = s가 되도록 해야함!

int n, s;
long long ans = 0;
int nums[40];
vector<int> l, r;

void l_dfs(int idx, int sum) {
	if (idx == n / 2) {
		l.push_back(sum);
		return;
	}

	// 여기 구문때문에 idx하나마다 *2의 시간복잡도.
	l_dfs(idx + 1, sum); // idx번째꺼 포함x
	l_dfs(idx + 1, sum + nums[idx]); // idx번째꺼 포함
}

void r_dfs(int idx, int sum) {
	// 끝까지 부분집합 다 체크한 base cond.
	if (idx == n) {
		r.push_back(sum);
		return;
	}

	//l_dfs와 마찬가지.
	r_dfs(idx + 1, sum);
	r_dfs(idx + 1, sum + nums[idx]);
}

int main() {
	cin.sync_with_stdio(false);
	cin.tie(NULL);
	cin >> n >> s;
	for (int i = 0; i < n; i++) {
		cin >> nums[i];
	}

	l_dfs(0, 0);
	r_dfs(n / 2, 0);

	sort(l.begin(), l.end()); // [0,1,2,2,4,5,6]
	sort(r.begin(), r.end(), greater<int>()); // [8,7,7,5,2,1,0]
#if DEB
	for (auto i : l)
		cout << i << ' ';
	cout << endl;
	for (auto i : r)
		cout << i << ' ';
	cout << endl;
#endif

	int ll = 0, rr = 0;
	int ls = l.size(), rs = r.size();
	while (ll < ls and rr < rs) {
		if (l[ll] + r[rr] == s) {
			long long p = 1, q = 1;
			ll++;
			rr++;
			// 같은수가 여러개 있을 경우의 계산.
			// 이렇게 한번에 안해주면 한 ll에 대해rr을 여러번 수행해야하는데
			// 이 과정에서 시간초과가 발생함.
			while (ll < ls and l[ll] == l[ll - 1]) {
				p++;
				ll++;
			}
			while (rr < rs and r[rr] == r[rr - 1]) {
				q++;
				rr++;
			}
			ans += p * q;
		}
		else if (l[ll] + r[rr] < s) {
			ll++;
		}
		else {
			rr++;
		}
	}


	// 아무 것도 고르지 않는 경우(l_dfs에서 위쪽만 선택, r_dfs에서 위쪽만 선택)일때 하나 빼야함
	// 양의 크기의 수열이라고 했으므로!
	if (s == 0) ans--;
	cout << ans;
	return 0;
}
#endif
```

#if 1
#define DEB 0
#include<iostream>
#include<queue>
using namespace std;

int n, num;

// ma는 median 이하의 maximum, mi는 median 초과의 minimum
priority_queue<int, vector<int>, less<int>> ma;
priority_queue<int, vector<int>, greater<int>> mi;
int main() {
	cin.sync_with_stdio(false);
	cin.tie(nullptr);
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> num;

		//first entry
		if (ma.empty()) {
			ma.push(num);
			cout << ma.top() << '\n';
			continue;
		}

		if (ma.size() > mi.size()) {
			if (num < ma.top()) {
				mi.push(ma.top());
				ma.pop();
				ma.push(num);
			}
			else {
				mi.push(num);
			}
		}

		// ma.size() == mi.size()
		else {
			if (num > mi.top()) {
				ma.push(mi.top());
				mi.pop();
				mi.push(num);
			}
			else {
				ma.push(num);
			}
		}
#if DEB
		cout << "ma.top(): " << ma.top() << ", ma.size(): " << ma.size() << endl;
		cout << "mi.top(): " << mi.top() << ", mi.size(): " << mi.size() << endl;
#endif
		cout << ma.top() << '\n';

	}
	return 0;

}
#endif

### 평범한 배낭
https://www.acmicpc.net/problem/12865

- DP. O(nk)
- 1차원으로는 이렇게함
```c++
#if 1
#include<iostream>
using namespace std;
int n, k, w, v;
int dp[100001]{ 0 };

// for i=0~n-1
//  for j=k~w
//   dp[j] = max(dp[j], dp[j-w[i]]+v[i])
// 아이템이 새로 업데이트될때마다, 최대부터 하나씩 줄여나가면서
// 들어갈수 있는데는 모조리 업데이트 해주는 방식. 이런 dp가 좀 어렵당.
int main() {
	cin.sync_with_stdio(false);
	cin.tie(nullptr);
	cin >> n >> k;
	for (int i = 0; i < n; i++) {
		cin >> w >> v;
		for (int j = k; j >= w; j--)
			if (dp[j] < dp[j - w] + v)
				dp[j] = dp[j - w] + v;
	}
	cout << dp[k];
	return 0;
}

#endif
```


```c++
#if 1
#include<iostream>
#include<vector>
#define max(x,y) x>y? x:y
using namespace std;
int n, k, w[101]{ 0 }, v[101]{ 0 };
int dp[101][100001]{ 0 };

// dp[i][j] : i번째까지 아이템을 넣을수있다고 할때, j만큼의 weight여유가있다고 할때의 최댓값.
// dp[i][j] = max(i번째 item이 들어갈 때: v[i]+dp[i-1][j-w[i]],
//					안들어갈때: dp[i-1][j] for i in range(0,n))

// 1차원으로 해보고싶은뎅


int main() {
	cin.sync_with_stdio(false);
	cin.tie(nullptr);
	cin >> n >> k;
	for (int i = 1; i <= n; i++)
		cin >> w[i] >> v[i];

	for (int i = 1; i <= n; i++) {
		for (int j = 0; j <= k; j++)
			dp[i][j] = dp[i - 1][j];

		for (int j = 0; j <= k; j++)
			if (j >= w[i])
				dp[i][j] = max(dp[i][j], dp[i - 1][j - w[i]] + v[i]);
	}

	cout << dp[n][k];

	return 0;
}

#endif
```
