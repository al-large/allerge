### 감소하는 수
https://www.acmicpc.net/problem/1038

- 수학, Backtracking
- Combination을 활용함. 9~0까지 수 중 r개를 뽑고 내림차순으로 정렬했을 때 i번째 수를 구하기 로 풀었다.
- 쉽게 푸는 방법도 무조건 있는데 난 걍 이렇게 풀어봤다.
- 토의를 통해 더 알아보도록 하좌.

```c++
#include<iostream>
#include<string>
using namespace std;

// 10Cr을 이용해 풀면 된다
// 9876543210 순으로 고르는게 순서가 맞음
// len이 정해지면, 그다음 10_C_len - idx - 1 번째 수를 고르면 된다.

int Combination(int n, int r) {
	if (r > n - r)
		return Combination(n, n - r);
	int ret = 1;
	for (int i = n; i > n - r; i--)
		ret *= i;
	for (int i = 1; i <= r; i++)
		ret /= i;
	return ret;
}
void dfs(int len, int len_idx, int& cnt, int num_idx, string& ret, bool visited[10]) {
	if (len_idx == len) {
		if (cnt == 0) {
			for (int i = 0; i <= 9; i++) {
				if (visited[i])
					ret += char(9 - i + '0');
			}
		}
		cnt--;
		return;
	}

	for (int i = num_idx; i <= 9; i++) {
		if (!visited[i]) {
			visited[i] = true;
			dfs(len, len_idx+1, cnt, i, ret, visited);
			visited[i] = false;
		}
	}
}

string DecreasingNumber(int len, int idx) {
	bool visited[10]{ false };
	int length = len, index = idx;
	string ret = "";
	// n_C_len 에서 idx번째 내뱉기?!?!
	dfs(length, 0, index, 0, ret, visited);
	return ret;
}

int main() {
	int n;
	cin >> n;
	//for (int i = 1; i <= 10; i++)
	//	cout << Combination(10, i) << endl;
	for (int i = 1; i <= 10; i++) {
		if (n < Combination(10, i)) {
			cout << DecreasingNumber(i, Combination(10, i) - n - 1)<<endl;
			return 0;
		}
		n -= Combination(10, i);
	}
	cout << -1 << endl;
	return 0;
}
```

### QR
https://www.acmicpc.net/problem/2680


```c++
#include<iostream>
#include<string>
using namespace std;
#define DEB 1
string alphanumeric = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ $%*+-./:";

int hex_to_int(char c) {
	if (c <= '9' && c >= '0') {
		return c - '0';
	}
	else {
		return c - 'A' + 10;
	}
}

string hex_to_bi(string s) {
	string ret = "";
	for (auto c : s) {
		string cc = "0000";
		unsigned int tmp = hex_to_int(c);
		for (int i = 0; i < 4; i++)
			cc[i] = ((tmp >> (3 - i)) & 1) + '0';
		ret += cc;
	}
	return ret;
}

int bi_to_int(string s) {
	int ret = 0;
	for (int i = s.size() - 1; i >= 0; i--) {
		ret += (s[i] - '0') << s.size() - i - 1;
	}
	return ret;
}


// stirng length == 6 or 11
string bi_to_alpha(string s) {
	string ret = "";
	if (s.size() == 11) {
		ret += alphanumeric[bi_to_int(s) / 45];
	}
	ret += alphanumeric[bi_to_int(s) % 45];

	return ret;
}


//string length == 8
string bi_to_hex(string s) {
	string ret = "";

	int n = bi_to_int(s.substr(0, 8));
	//print-able.
	if (n >= 16 * 2 && n <= 16 * 7 + 15) {
		if (char(n) == '\\' || char(n) == '#')
			ret += '\\';
		return ret + char(n);
	}

	//not print-able
	ret += '\\';
	while (s.size() >= 4) {
		n = bi_to_int(s.substr(0, 4));
		s = s.substr(4);
		if (n >= 10) {
			ret += char(n - 10 + 'A');
		}
		else {
			ret += char(n + '0');
		}
	}

	return ret;
}

//string length == 13
string bi_to_kanji(string s) {
	string ret = "";
	int n = bi_to_int(s.substr(0, 13));

	//print-able.
	if (n >= 16 * 2 && n <= 16 * 7 + 15) {
		if (char(n) == '\\' || char(n) == '#')
			ret += '\\';
		return ret + char(n);
	}

	//not print-able
	ret += '#';
	ret += s[0];
	s = s.substr(1);
	while (s.size() >= 4) {
		int n = bi_to_int(s.substr(0, 4));
		s = s.substr(4);
		if (n >= 10) {
			ret += char(n - 10 + 'A');
		}
		else {
			ret += char(n + '0');
		}
	}
	return ret;
}

string decode(int& data_cnt, string s) {
	string ret = "";
	if (s.length() < 4)
		return ret;

	string mode = s.substr(0, 4);
	s = s.substr(4);

	//numeric
	if (mode == "0001") {
		int cnt = bi_to_int(s.substr(0, 10));
		data_cnt += cnt;
		s = s.substr(10);
		while (cnt >= 3) {
			ret += to_string(bi_to_int(s.substr(0, 10)));
			s = s.substr(10);
			cnt -= 3;
		}

		if (cnt == 2) {
			ret += to_string(bi_to_int(s.substr(0, 7)));
			s = s.substr(7);
			cnt -= 2;
		}
		else if (cnt == 1) {
			ret += to_string(bi_to_int(s.substr(0, 4)));
			s = s.substr(4);
			cnt -= 1;
		}
	}

	//alphanumeric
	else if (mode == "0010") {
		int cnt = bi_to_int(s.substr(0, 9));
		data_cnt += cnt;
		s = s.substr(9);
		while (cnt >= 2) {
			ret += bi_to_alpha(s.substr(0, 11));
			s = s.substr(11);
			cnt -= 2;
		}

		if (cnt == 1) {
			ret += bi_to_alpha(s.substr(0, 6));
			s = s.substr(6);
			cnt -= 1;
		}
	}

	// 8 bit byte
	else if (mode == "0100") {
		int cnt = bi_to_int(s.substr(0, 8));
		data_cnt += cnt;
		s = s.substr(8);
		while (cnt) {
			ret += bi_to_hex(s.substr(0, 8));
			s = s.substr(8);
			cnt -= 1;
		}
	}

	// kanji
	else if (mode == "1000") {
		int cnt = bi_to_int(s.substr(0, 8));
		data_cnt += cnt;
		s = s.substr(8);
		while (cnt) {
			ret += bi_to_kanji(s.substr(0, 13));
			s = s.substr(13);
			cnt -= 1;
		}
	}

	//termination
	else {
		return ret;
	}

	//goes next!
	return ret + decode(data_cnt, s);
}

int main() {
#if DEB

	s = "010000000011010001011001001010100011";
	cout << decode(data_cnt, s) << endl;

	cout << decode(data_cnt, s) << endl;
	s = "1000000001001111111111111111111110000100000000000110000001001000";
	cout << decode(data_cnt, s) << endl;
#endif

#if !DEB
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	int n;
	cin >> n;
	while (n--) {
		int data_cnt = 0;
		string a;
		cin >> a;
		a = hex_to_bi(a);
		a = decode(data_cnt, a);
		cout << data_cnt << " " << a << endl;
	}
#endif
	return 0;
}
```

### 79. Word Search
https://leetcode.com/problems/word-search/

- `return` 이 있는 DFS의 경우, 방문 후 함수 진입이 아니라 함수 진입 후 방문으로 짜야 편하다!
- 그 연습을 할 수 있어 좋았다. 전방향 탐색이므로 `ret |= ...` 부분도 좋은 활용이다.
- 근데 다른 사람들이랑 logic이 완전 같은데 TLE가 난다. 아마 TC가 추가된 것 같은데, 다른 사람 코드를 써도 TLE가 나는 것을 보니 다른 방향으로 짜는걸 시도해봐야겠다.

```c++
int d[5] = { -1,0,1,0,-1 };
class Solution {
public:
	bool dfs(int x, int y, int idx, vector<vector<char>>& board, string word) {
		if (word[idx] == '\0')
			return true;
		int n = board.size();
		int m = board[0].size();
		if (x < 0 || x >= n || y < 0 || y >= m)
			return false;
		if (board[x][y] != word[idx])
			return false;

		board[x][y] = '*';

		bool ret = false;
		for (int i = 0; i < 4; i++) {
			int nx = x + d[i], ny = y + d[i + 1];
			ret |= dfs(nx, ny, idx + 1, board, word);
		}

		board[x][y] = word[idx];
		return ret;
	}
	bool exist(vector<vector<char>>& board, string word) {
		if (board.empty() || word.empty())
			return false;
		else if (board[0].empty())
			return false;

		// backtracking은 기본적으로 dfs로 합시다!
		int n = board.size();
		int m = board[0].size();
		for (int x = 0; x < n; x++) {
			for (int y = 0; y < m; y++) {
				if (dfs(x, y, 0, board, word))
					return true;
			}
		}
		return false;
	}
};
```

### 648. Replace Words
https://leetcode.com/problems/replace-words/

튜라이 튜라이 외워버렸지 뭐얌

const char쓰다가 string으로 넘어가버렸지뭐얌

하지만 포인터가 좀더 효율적일 것 같다는 생각이 들지 뭐얌


```c++
#include<iostream>
#include<vector>
#include<string>
using namespace std;

class Trie {
public:
	bool isTerminal;
	Trie* child[26];

	Trie() {
		isTerminal = false;
		for (int i = 0; i < 26; i++)
			child[i] = NULL;
	}

	~Trie() {
		for (int i = 0; i < 26; i++)
			delete child[i];
	}

	void insert(string s, int idx = 0) {
		if (s[idx] == '\0')
			isTerminal = true;

		else {
			if (child[s[idx] - 'a'] == NULL)
				child[s[idx] - 'a'] = new Trie();

			child[s[idx] - 'a']->insert(s, idx + 1);
		}
	}

	//// 이게 원형
	//Trie* find(string s, int idx) {
	//	if (s[idx] == '\0' && isTerminal)
	//		return this;
	//	if (child[s[idx] - 'a'] == NULL)
	//		return NULL;

	//	return child[s[idx] - 'a']->find(s, idx + 1);
	//}

	//이건 변형
	void replace(string& s, int idx = 0) {
		if (s[idx] == '\0')
			return;
		else if (isTerminal)
			s = s.substr(0, idx);
		else if (child[s[idx] - 'a'])
			child[s[idx] - 'a']->replace(s, idx + 1);
	}
};

class Solution {
public:
	string replaceWords(vector<string>& dictionary, string sentence) {
		Trie* root = new Trie();
		for (auto s : dictionary)
			root->insert(s);

		int i = 0;
		string ret = "";
		string tmp = "";
		while (sentence[i] != '\0') {
			if (sentence[i] != ' ')
				tmp += sentence[i];
			else if (!tmp.empty()) {
				root->replace(tmp);
				ret += tmp + " ";
				tmp = "";
			}
			i++;
		}

		// for the last word.
		root->replace(tmp);
		ret += tmp;
		return ret;
	}
};
```
